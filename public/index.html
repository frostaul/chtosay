<script>

    const tg = window.Telegram.WebApp;

    tg.ready();

    tg.expand();


    const SHEET_ID = '1yIwxqPfZCDbnU9PWd2seC2chdVjhWLH22JCP34cRv74';

    let allData = [];

    let currentRating = '';

    let lastType = '', lastRating = '', lastDuration = '';


    async function load() {

        try {

            const r = await fetch(`https://docs.google.com/spreadsheets/d/${SHEET_ID}/export?format=csv&gid=0`);

            const text = await r.text();

            const rows = text.split('\n').slice(1);

            allData = rows.map(line => {

                const cols = line.split(',');

                return {

                    type: (cols[0]||'').trim().toLowerCase(),

                    rating: (cols[1]||'').trim().toLowerCase(),

                    duration: parseInt(cols[2])||0,

                    title: (cols[3]||'').trim(),

                    link: (cols[4]||'').trim()

                };

            }).filter(item => item.type);

            document.getElementById('loading').classList.add('hidden');

            go('start');

        } catch(e) {

            document.getElementById('loading').classList.add('hidden');

            document.getElementById('error').classList.remove('hidden');

        }

    }


    function go(screen, rating) {

        tg.HapticFeedback?.impactOccurred('light');

        if(rating) currentRating = rating;

        document.querySelectorAll('#start,#film,#filmDuration,#serial,#serialDuration,#random,#result')

            .forEach(el => el.classList.add('hidden'));

        document.getElementById(screen).classList.remove('hidden');

        if(screen==='start') tg.BackButton?.hide();

        else {

            tg.BackButton?.show();

            tg.BackButton?.onClick(() => {

                if(screen==='filmDuration' || screen==='serialDuration') go(currentRating? (lastType==='film'?'film':'serial') : 'start');

                else if(screen==='result') go(lastType==='film'?'filmDuration':'serialDuration');

                else go('start');

            });

        }

    }


    function show(type, rating, duration) {

        lastType=type; lastRating=rating; lastDuration=duration;

        tg.HapticFeedback?.impactOccurred('medium');

        let filtered = allData.filter(i => i.type===type && i.rating===rating);

        if(type==='film') {

            if(duration==='short') filtered=filtered.filter(i => i.duration <= 119);

            else if(duration==='long') filtered=filtered.filter(i => i.duration >= 120);

        } else {

            if(duration==='short') filtered=filtered.filter(i => i.duration <= 35);

            else if(duration==='long') filtered=filtered.filter(i => i.duration >= 36);

        }

        display(filtered);

    }


    function randomPick(rating) {

        lastType = Math.random()>0.5?'film':'serial';

        lastRating = rating;

        lastDuration = 'any';

        tg.HapticFeedback?.impactOccurred('medium');

        display(allData.filter(i => i.rating===rating));

    }


    function display(filtered) {

        go('result');

        if(filtered.length>0) {

            const pick = filtered[Math.floor(Math.random()*filtered.length)];

            document.getElementById('resTitle').textContent = (lastType==='film'?'🎬 Фильм':'📺 Сериал');

            document.getElementById('resCard').innerHTML = `

                <div class="card">

                    <span class="badge badge-${lastRating}">${lastRating.toUpperCase()}</span>

                    <h3>${pick.title}</h3>

                    <p class="inf">⏱ ${pick.duration} мин</p>

                    <a href="${pick.link}" target="_blank">Смотреть на Кинопоиске →</a>

                </div>`;

        } else {

            document.getElementById('resTitle').textContent = '😔';

            document.getElementById('resCard').innerHTML = '<div class="card"><p>Ничего не найдено</p></div>';

        }

    }


    function retry() {

        if(lastDuration==='any') randomPick(lastRating);

        else show(lastType, lastRating, lastDuration);

    }


    function openLink(url) { tg.openLink?.(url) || window.open(url,'_blank'); }


    load();

</script>
